addXYZ :: Nat -> Nat -> Nat -> Stream ()
addXYZ x y z =
  case x of
    O | y == z -> return ()
      | otherwise -> Empty
    S x' ->
      case z of
        O -> Empty
        S z' -> addXYZ x' y z'
