let rec mult$^o$ x y z = conde [
  (x === O /\ z === O);
  (y === O /\ z === O);
  (x === S O /\ z === y);
  (y === S O /\ z === x);
  (fresh (x' r')
    (x === S x') /\ (add y r' z) /\ (mult x' y r')
  )]
