multXY :: Nat -> Nat -> Stream Nat
multXY O      y     = return O
multXY x      O     = return O
multXY (S O)  y     = return y
multXY x      (S O) = return x
multXY (S x') y     = do
  r' <- multXY x' y
  addXY y r'
