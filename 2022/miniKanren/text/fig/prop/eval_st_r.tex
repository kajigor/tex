evalStR :: [Bool] -> Bool -> Stream Term
evalStR st result =
      lit  st result `mplus`
      var  st result `mplus`
      neg  st result `mplus`
      disj st result `mplus`
      conj st result
  where
    conj st result = do
      (v, w) <- andR result
      y <- evalStR st w
      x <- evalStR st v
      return (Conj x y)
    disj st result = do
      (v, w) <- orR result
      y <- evalStR st w
      x <- evalStR st v
      return (Disj x y)
    neg st result = do
      v <- notR result
      x <- evalStR st v
      return (Neg x)
    var st result = do
      z <- elemStR st result
      return (Var z)
    lit st result =
      if result
      then return (Lit True)
      else return (Lit False)
    \end{lstlisting}
