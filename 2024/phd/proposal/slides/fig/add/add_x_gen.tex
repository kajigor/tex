 addIOO $::$ Nat -> Stream Nat -> Stream (Nat, Nat)
 addIOO x gen$_y$ =
   case x of
     O -> do
       y <- gen$_y$
       return (y, y)
     S x$_1$ -> do
       (y, z$_1$) <- addIOO x$_1$ gen$_y$
       return (y, S z$_1$)
