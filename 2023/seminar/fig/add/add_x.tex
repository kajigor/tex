addIOO :: Nat -> Stream (Nat, Nat)
addIOO x = case x of
           O -> do
             z <- genNat
             return (z, z)
           S x' -> do
             (y, z') <- addIOO x'
             return (y, S z')

genNat :: Stream Nat
genNat = Mature O (S <$\$$> genNat)
