 ~\texttt{muloIIO $::$ Nat $\to$ Nat $\to$ Stream Nat}~
 muloIIO x y =
     zero <$\mid$> positive
   where
     zero = do
       guard (x == O)
       return O
     positive = do
       (S $x_1$) <- return x
       $z_1$ <- muloIIO $x_1$ y
       z <- addoIIO $z_1$ y
       return z
