addOOI :: Nat -> Stream (Nat, Nat)
addOOI z =
  return (O, z) `mplus`
  case z of
    O -> Empty
    S z$_1$ -> do
      (x$_1$, y) <- addOOI z$_1$
      return (S x$_1$, y)
