 addIOO $::$ Nat -> Stream (Nat, Nat)
 addIOO x =
   case x of
     O -> do
       z <- genNat
       return (z, z)
     S x$_1$ -> do
       (y, z$_1$) <- addIOO x$_1$
       return (y, S z$_1$)

 genNat $::$ Stream Nat
 genNat = Mature O (S <$\$$> genNat)
