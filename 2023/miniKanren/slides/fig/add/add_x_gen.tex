addIOO :: Nat -> Stream Nat -> Stream (Nat, Nat)
addIOO x gen$_z$ = case x of
  O -> do
    z <- gen$_z$
    return (z, z)
  S x$_1$ -> do
    (y, z$_1$) <- addIOO x$_1$ gen$_z$
    return (y, S z$_1$)
