multOIO :: Nat -> Stream Nat -> Stream Nat
multOIO y gen_add$_z$ =
    return (O, O) `mplus`
    do
        (z$_1$, z) <- addIOO y gen_add$_z$
        x <- muloOII y z$_1$
        return (S x, z)
